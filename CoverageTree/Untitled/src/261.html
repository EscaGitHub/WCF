using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using Swr.Configurator.Data;
using Swr.Configurator.Data.ProductModel;

namespace Swr.Configurator.Logic.EventArgs
{
    public class ProductEventArgs:System.EventArgs
    {

        public ProductEventArgs(List<IProduct> products, bool fromSearch)
        {
            Products = products;
            FromSearch = fromSearch;
        }
        public List<IProduct> Products { get; set; }

        public bool FromSearch { get; set; }
    }
}
