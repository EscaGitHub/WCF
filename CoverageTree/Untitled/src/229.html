using System;
using System.Globalization;
using System.Windows.Data;
using SWR_Tools.Extention;

namespace Swr.Configurator.View.Logic
{
	public class KeyTipFromCommandConverter : IValueConverter
	{
		public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
		{
			var command = value as ConfiguratorRelayCommand;
			if (command == null)
				return null;

			if (ConfiguratorMainWindowVM.Instance == null)
				return null;

			var settings = ConfiguratorMainWindowVM.Instance.LocalSettingsController.Settings;
			if (settings == null)
				return null;

			var shortcuts = settings.HotKeysMapping;
			if (!shortcuts.SafeAny())
				return null;

			var shortcut = shortcuts[command.Type];
			if (shortcut == null)
				return null;

			return shortcut.Gesture;
		}

		public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
		{
			return Binding.DoNothing;
		}
	}
}
