using System;
using System.Collections.Generic;
using System.Windows.Media;
using System.Xml.Serialization;
using Newtonsoft.Json;

namespace Swr.Configurator.Common.Data.ObjectModel
{
    /// <summary>
    /// Настройки компонента
    /// </summary>
    public class ComponentTypeObject: IObjectStatus
    {
        public ComponentTypeObject()
        {
            Attributes = new List<AttributeForTypeObject>();
            SpecificationTypeNames = new List<string>();
        }
        /// <summary>
        /// Идентификатор
        /// </summary>
        public Guid ID { get; set; }

        /// <summary>
        /// Имя типа компонентов
        /// </summary>
        public string Name { get; set; }

        /// <summary>
        /// Маска имени
        /// </summary>
        public string ComponentNameTemplate { get; set; }

        /// <summary>
        /// Список атрибутов
        /// </summary>
        public List<AttributeForTypeObject> Attributes { get; set; }

        /// <summary>
        /// Иконка в виде строки Base64
        /// </summary>
        public string ElementIcon { get; set; }

        public bool IsNew { get; set; }

        [XmlIgnore]
        public ImageSource ImageIcon { get; set; }

        /// <summary>
        /// Порядок сортировки
        /// </summary>
        public int SortOrder { get; set; }

        /// <summary>
        /// Разрешено ли создавать структуру
        /// </summary>
        //public bool AllowStructure { get; set; }

        public List<string> SpecificationTypeNames { get; set; }

        public void Dispose()
        {
            if (Attributes != null)
            {
                foreach (var attributeForTypeObject in Attributes)
                {
                    attributeForTypeObject.Dispose();
                }

                Attributes.Clear();
                Attributes = null;
            }

            if (SpecificationTypeNames != null)
            {
                SpecificationTypeNames.Clear();
                SpecificationTypeNames = null;
            }

            ImageIcon = null;
        }
    }
}
