using System;
using System.Collections;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using Swr.Configurator.View.DataViewModels;

namespace Swr.Configurator.Logic
{
	public class ProductSorter : IComparer
	{
		public int Compare(object x, object y)
		{
			var productVM1 = x as ProductVM;
			var productVM2 = y as ProductVM;

			if (productVM1 == null)
				return -1;

			if (productVM2 == null)
				return 1;

			var product1 = productVM1.Product;
			var product2 = productVM2.Product;

			if (product1.Type == null)
				return -1;

			if (product2.Type == null)
				return 1;

			if (product1.Type.SortOrder < product2.Type.SortOrder)
				return -1;
			if (product1.Type.SortOrder > product2.Type.SortOrder)
				return 1;

			return string.Compare(product1.Name, product2.Name, StringComparison.OrdinalIgnoreCase);
		}
	}
}
