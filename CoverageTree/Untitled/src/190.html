using System;
using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;
using System.Text;

namespace Swr.Configurator.Common.Data.Api
{
	public class SpecificationComponentApi
	{
        /// <summary>
        /// Идентификатор компонента, входящего в спецификацию
        /// </summary>
        [CannotBeEmptyGuid(ErrorMessage = "Field 'Id' is required.")]
        public Guid ID { get; set; }

		/// <summary>
		/// Атрибуты связи для компонента
		/// </summary>
		public List<AttributeApi> RelationAttributes { get; set; }

        /// <summary>
        /// Атрибуты компонента
        /// </summary>
        public List<AttributeApi> ComponentAttributes { get; set; }

        /// <summary>
        /// Имя типа компонента
        /// </summary>
        public string TypeName { get; set; }

        /// <summary>
        /// Признак, удален ли компонент из спецификации
        /// </summary>
        public bool IsDeleted { get; set; }

        public bool HasChildren { get; set; }

        public string SpecificationTypeName { get; set; }
	}
}
