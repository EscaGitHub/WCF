using System.Collections.Generic;
using System.Windows.Input;

namespace Swr.Configurator.View.Controls
{
	public class KeysToIgnore
	{
		private static List<Key> _keys;

		public static List<Key> Keys
		{
			get
			{
				return _keys ?? (_keys = new List<Key>
				{
					Key.Left, Key.Right, Key.Up, Key.Down, Key.LeftShift, Key.RightShift, Key.Delete, Key.Insert,
					Key.PageDown, Key.PageUp, Key.LeftAlt, Key.RightAlt, Key.LeftCtrl, Key.RightCtrl, Key.Tab,
					Key.F1, Key.F3, Key.F4, Key.F5, Key.F6, Key.F7, Key.F8, Key.F9, Key.F10, Key.F11, Key.F12,
					Key.Home, Key.End, Key.Clear, Key.PrintScreen, Key.Sleep, Key.LWin, Key.RWin, Key.CapsLock
				});
			}
		}
	}
}
