using System.Collections.Generic;
using System.ComponentModel;

namespace Swr.Configurator.Common.Data
{
    public class SearchParams
    {
        public SearchParams()
        {
            SearchAttributes = new List<SearchParam>();
        }

        /// <summary>
        /// Максимальное количество элементов в результате
        /// </summary>
        public int Count { get; set; }

        /// <summary>
        /// Номер страницы поиска
        /// </summary>
        public int PageNumber { get; set; }

        [CannotBeEmpty(ErrorMessage = "Filed 'SearchAttributes' is required.")]
        public List<SearchParam> SearchAttributes { get; set; }
    }
}
