using System;
using System.Globalization;
using System.Linq;
using System.Windows.Data;

namespace Swr.Configurator.View.Logic
{
	public class MouseOverDeletedProductConverter : IMultiValueConverter
	{
		public object Convert(object[] values, Type targetType, object parameter, CultureInfo culture)
		{
			if (values == null || values.Count() != 2 || !(values[0] is bool) || !(values[1] is bool))
				return false;
			var isDeleted = (bool) values[0];
			var isMouseOver = (bool)values[1];

			return isDeleted && isMouseOver;
		}

		public object[] ConvertBack(object value, Type[] targetTypes, object parameter, CultureInfo culture)
		{
			return new[] {Binding.DoNothing, Binding.DoNothing};
		}
	}
}
