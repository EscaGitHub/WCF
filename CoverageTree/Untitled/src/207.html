using Swr.Configurator.Data.SpecificationModel;

namespace Swr.Configurator.Data.EventArgs
{
	/// <summary>
	/// ��������� ������� ��������� ��������
	/// </summary>
	public class ReferencePropertyChangedEventArgs : System.EventArgs
	{
		public ReferencePropertyChangedEventArgs(IReference reference, string variable, string oldValue, string newValue)
		{
			Reference = reference;
			Variable = variable;
			OldValue = oldValue;
			NewValue = newValue;
		}

		public IReference Reference { get; private set; }

		public string Variable { get; private set; }

		public string OldValue { get; private set; }

		public string NewValue { get; private set; }
	}
}
