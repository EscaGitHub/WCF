using System;
using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;
using System.Text;

namespace Swr.Configurator.Common.Data
{
    [AttributeUsage(AttributeTargets.Property)]
    class CannotBeEmptyGuidAttribute : RequiredAttribute
    {
        public override bool IsValid(object value)
        {
            if (value != null)
            {
                var guid = value.ToString();

                if (Guid.TryParse(guid, out var result))
                {
                    if (result != Guid.Empty) return true;
                }
            }

            return false;
        }
    }
}
