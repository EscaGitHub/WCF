using System;
using System.Globalization;
using System.Linq;
using System.Windows.Controls;
using System.Windows.Data;

namespace Swr.Configurator.View.Logic
{
	public class IsLockedCellConverter : IMultiValueConverter
	{
		public object Convert(object[] values, Type targetType, object parameter, CultureInfo culture)
		{
			if (values == null || values.Count() != 3 || !(values[1] is DataGridColumn))
				return null;

			var column = values[1] as DataGridColumn;

			return CellLockStatusChecker.IsLocked(column, values[0]);
		}

		public object[] ConvertBack(object value, Type[] targetTypes, object parameter, CultureInfo culture)
		{
			return new[] {Binding.DoNothing, Binding.DoNothing, Binding.DoNothing};
		}
	}
}
