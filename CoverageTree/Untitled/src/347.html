using Microsoft.VisualStudio.TestTools.UnitTesting;
using Swr.Configurator.Logic;

namespace Swr.Configurator.Test.Net
{
    [TestClass]
    public class MaskNameParserTest
    {
        [TestMethod]
        [DataRow("%обозначение% - %Наименование%", "наименование", true)]
        [DataRow("%обозначение% - %Наименование%", "Наименование", true)]
        [DataRow("%обозначение% - %Наимено вание%", "наимено вание", true)]
        [DataRow("%% - %Наименование%", "наименование", true)]
        [DataRow("%обозначение% - %Наименование11%", "наименование11", true)]
        [DataRow("%обозначение% - %Наименование%", "Примечание", false)]
        [DataRow("%обозначение% - %Наименование.%", "наименование", false)]
        public void RegexpTest(string mask, string property, bool resultRow)
        {
            var result = MaskNameParser.PropertyInMask(mask, property);

            Assert.AreEqual(result, resultRow);
        }
    }
}
