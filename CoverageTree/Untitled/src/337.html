using System;
using Swr.Configurator.Data;
using SwrUILogic;

namespace Swr.Configurator.View.Logic
{
	public class ConfiguratorRelayCommand : RelayCommand
	{
		public ConfiguratorRelayCommand(Action<object> execute, ConfiguratonCommand commandType, string name = "", string description = "", string defaultShortcut = "", CommandGroup group = CommandGroup.Other)
			: base(execute)
		{
			Type = commandType;
			Name = name;
			Description = description;
			DefaultShortcut = defaultShortcut;
		}

		public ConfiguratorRelayCommand(Action<object> execute, Predicate<object> canExecute, ConfiguratonCommand commandType, string name = "",
			string description = "", string defaultShortcut = "", string icon = "", CommandGroup group = CommandGroup.Other, int top = 0, int left = 0)
			: base(execute, canExecute)
		{
			Type = commandType;
			Name = name;
			Description = description;
			DefaultShortcut = defaultShortcut;
			IconData = icon;
			Group = group;
			Top = top;
			Left = left;
		}

		public CommandGroup Group { get; set; }

        /// <summary>
        /// ������ ������ ��� �������� ������
        /// </summary>
		public int Top { get; set; }

	    /// <summary>
	    /// ������ ����� ��� �������� ������
	    /// </summary>
		public int Left { get; set; }

		public ConfiguratonCommand Type { get; private set; }

		public string Name { get; private set; }

		public string Description { get; private set; }

		public string DefaultShortcut { get; private set; }

		public string IconData { get; private set; }
	}

	public enum CommandGroup
	{
		Other,
		Specification,
		Order,
		Structure
	}
}
