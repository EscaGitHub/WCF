using System.ComponentModel;
using System.Runtime.CompilerServices;
using System.Windows;
using System.Windows.Controls;
using Swr.Configurator.Properties;
using Swr.Configurator.View.DataViewModels;

namespace Swr.Configurator.View.Controls
{
	/// <summary>
	/// Логика взаимодействия для ComponentTreeUC.xaml
	/// </summary>
	public partial class ComponentTreeUC : UserControl, INotifyPropertyChanged
	{
		public ComponentTreeUC()
		{
			InitializeComponent();
		}

		private StructureVM vm;

		private void StructureUC_OnDataContextChanged(object sender, DependencyPropertyChangedEventArgs e)
		{
			vm = e.NewValue as StructureVM;
		}

		public event PropertyChangedEventHandler PropertyChanged;

		protected virtual void OnPropertyChanged([CallerMemberName] string propertyName = null)
		{
			PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(propertyName));
		}
	}
}
