using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Text.RegularExpressions;
using System.Threading.Tasks;

namespace Swr.Configurator.Logic
{
    public static class MaskNameParser
    {
        public static bool PropertyInMask(string mask, string propertyName)
        {
            const string pattern = "%[\\w\\s]+%";

            var regex = new Regex(pattern, RegexOptions.None,TimeSpan.FromMilliseconds(1000));

            var match = regex.Match(mask);

            while (match.Success)
            {
                var value = match.Groups[0].Value;

                value = value.Replace("%", string.Empty);

                if (string.Compare(value, propertyName, StringComparison.InvariantCultureIgnoreCase) == 0) return true;

                match = match.NextMatch();
            }

            return false;
        }
    }
}
