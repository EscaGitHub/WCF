using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows;
using System.Windows.Data;
using Swr.Configurator.View.DataViewModels;

namespace Swr.Configurator.View.Controls
{
	public class StructureTreeUC : ComponentTreeUC
	{
		public StructureTreeUC()
		{
			Binding b = new Binding("DataContext")
			{
				Source = ConfiguratorMainWindowVM.Instance,
				Path = new PropertyPath("StructureVM"),
				Mode = BindingMode.OneWay
			};
			this.SetBinding(DataContextProperty, b);
		}
	}
}
