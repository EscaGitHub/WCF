using System;
using System.Collections.Generic;
using System.Text;

namespace Swr.Configurator.Common.Data.ObjectModel
{
    /// <summary>
    /// Тип спецификации
    /// </summary>
    public class SpecificationTypeObject : IObjectStatus
    {
        public SpecificationTypeObject()
        {
            Attributes = new List<AttributeForTypeObject>();
            AllowComponentTypeNames = new List<string>();
        }
        /// <summary>
        /// Идентификатор
        /// </summary>
        public Guid ID { get; set; }

        /// <summary>
        /// Имя типа спецификации
        /// </summary>
        public string Name { get; set; }

        /// <summary>
        /// Маска имени спецификации
        /// </summary>
        public string NameTemplate { get; set; }

        /// <summary>
        /// Список атрибутов
        /// </summary>
        public List<AttributeForTypeObject> Attributes { get; set; }

        public bool IsNew { get; set; }

        /// <summary>
        /// Список разрешенных типов компонентов
        /// </summary>
        public List<string> AllowComponentTypeNames { get; set; }

        public void Dispose()
        {
            if (Attributes != null)
            {
                foreach (var attributeForTypeObject in Attributes)
                {
                    attributeForTypeObject.Dispose();
                }

                Attributes.Clear();
                Attributes = null;
            }

            if (AllowComponentTypeNames != null)
            {
                AllowComponentTypeNames.Clear();
                AllowComponentTypeNames = null;
            }

        }
    }
}
