using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;

namespace Swr.Configurator.Common.Data.Api
{
    public class ComponentApiNew : IComponentApi
    {
        public ComponentApiNew()
        {
            Attributes = new List<AttributeApi>();
        }

        [Required(ErrorMessage = "Field 'TypeName' is required.")]
        public string TypeName { get; set; }

        [CannotBeEmpty(ErrorMessage = "Field 'Attributes' is required.")]
        public List<AttributeApi> Attributes { get; set; }
        public int Counter { get; set; }
    }
}
