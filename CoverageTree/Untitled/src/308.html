using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Data;
using Swr.Configurator.Common.Settings.Lists;

namespace Swr.Configurator.View.Logic
{
    public class PropertyValueListRecordConverter : IValueConverter
    {
        public ListSettings Settings { get; set; }

        public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
        {
            return value == null ? null : value.ToString();
        }

        public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
        {
            var record = value as ListRecordContent;

            if (record == null)
                return value;

            return record.Value;
        }
    }
}
