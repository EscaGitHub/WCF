using System;
using Swr.Configurator.Logic.Commands;

namespace Swr.Configurator.View.Commands
{
	public abstract class ViewCommandBase : IDisposable
	{
		protected ICommandManager _commandManager;

		protected ViewCommandBase(ICommandManager commandManager)
		{
			if (commandManager == null) throw new ArgumentNullException("commandManager");

			_commandManager = commandManager;
		}

		/// <summary>
		/// ������� ���������� ���������� �������
		/// </summary>
		public event EventHandler<CommandFinishedEventArgs> Finished;

		protected void OnFinished(bool result)
		{
			EventHandler<CommandFinishedEventArgs> handler = Finished;
			if (handler != null) handler(this, new CommandFinishedEventArgs(result));
		}

		public virtual void Dispose()
		{
		}

		public abstract bool Execute();
	}

	public class CommandFinishedEventArgs : EventArgs
	{
		public bool Result { get; set; }

		public CommandFinishedEventArgs(bool result)
		{
			Result = result;
		}
	}
}
