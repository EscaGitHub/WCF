using System;
using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;

namespace SWR.ConfiguratorApi.Data.Db
{
	public class SpecificationType
	{
        public SpecificationType()
        {
            SpecificationTypeAttributes = new List<SpecificationTypeAttribute>();
            AllowComponentTypes = new List<string>();
        }
		public Guid ID { get; set; }

		[MaxLength(255)]
		public string Name { get; set; }

        public List<SpecificationTypeAttribute> SpecificationTypeAttributes { get; set; }

        /// <summary>
        /// Список разрешенных типов компонентов
        /// </summary>
        public List<string> AllowComponentTypes { get; set; }
	}
}
