using System;
using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;
using System.Text;

namespace Swr.Configurator.Common.Data.Api
{
	public class SpecificationApi
	{
        public SpecificationApi()
        {
            Childs = new List<SpecificationComponentApi>();
        }
        /// <summary>
        /// Уникальный идентификатор компонента, для которого требуется создать спецификацию
        /// </summary>
        [CannotBeEmptyGuid(ErrorMessage = "Field 'Id' is required.")]
        public Guid ComponentID { get; set; }


        /// <summary>
        /// Имя типа спецификации
        /// </summary>
        [Required(ErrorMessage = "Field 'TypeName' is required.")]
        public string TypeName { get; set; }

        /// <summary>
        /// Компоненты, входящие в спецификацию
        /// </summary>
        [CannotBeEmpty(ErrorMessage = "Field 'Childs' is required.")]
        public List<SpecificationComponentApi> Childs { get; set; }
	}
}
