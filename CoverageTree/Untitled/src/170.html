using System;
using Swr.Configurator.Common.Data;

namespace Swr.Configurator.Common.Logic
{
    public class EntityChangedSuspender : IDisposable
    {
        private readonly IChangable _changable;
        private readonly bool _oldState;

        public EntityChangedSuspender(IChangable changable)
        {
            _changable = changable ?? throw new ArgumentNullException("changable");

            _oldState = _changable.SuspendChanged;
            _changable.SuspendChanged = true;
        }

        /// <summary>
        /// Performs application-defined tasks associated with freeing, releasing, or resetting unmanaged resources.
        /// </summary>
        /// <filterpriority>2</filterpriority>
        public void Dispose()
        {
            _changable.SuspendChanged = _oldState;
        }
    }
}
