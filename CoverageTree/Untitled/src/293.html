using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using Swr.Configurator.Data;
using Swr.Configurator.Data.ProductModel;
using Swr.Configurator.Logic.EventArgs;
using Swr.Configurator.View;

namespace Swr.Configurator.Logic
{
    public class ProductStructureContainer 
    {
        private readonly List<IProduct> _products = new List<IProduct>();

        public void AddProduct(IProduct product, bool fromSearch)
        {
            _products.Add(product);
            OnProductAdded(product, fromSearch);
        }

        public IEnumerable<IProduct> Products
        {
            get { return _products; }
        }

        public event EventHandler<ProductEventArgs> ProductAdded;

        private void OnProductAdded(IProduct product, bool fromSearch)
        {
            var handle = ProductAdded;
            handle?.Invoke(this, new ProductEventArgs(new List<IProduct> {product}, fromSearch));

        }


        public void CloseProduct(List<IProduct> products)
        {
            foreach (var product in products)
            {
                _products.Remove(product);
            }
            OnProductClosed(products);

        }

        public event EventHandler<ProductEventArgs> ProductClosed;

        private void OnProductClosed(List<IProduct> products)
        {
            var handle = ProductClosed;
            handle?.Invoke(this, new ProductEventArgs(products, false));

        }
    }
}
