namespace Swr.Configurator.View.Controls
{
	/// <summary>
	/// Interaction logic for SearchComponentControl.xaml
	/// </summary>
	public partial class SearchComponentControl : UserControl
	{
		public SearchComponentControl()
		{
			InitializeComponent();
			Vm = new SearchCompomentVm();
			DataContext = Vm;
            
		}

        private SearchCompomentVm Vm { get; set; }

        public CardController CardController { get; set; }

        public void SetProductStructure(ProductStructureContainer productStructureContainer)
        {
            if (Vm != null)
            {
                Vm.ProductStructureContainer = productStructureContainer;
                
            }
        }

        private void Selector_OnSelected(object sender, RoutedEventArgs e)
        {
            if (Vm?.SelectedItem is Product selectedItem)
            {
	            CardController.ShowCardItem(selectedItem, null, this);

            }
        }
	}
}
