using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using Swr.Configurator.Data.EventArgs;
using SwrUI;
using SwrUILogic.Data;

namespace Swr.Configurator.View.Settings
{
    public class Separator : PageControlSeparator, IValidationUpdate
    {
        public void UpdateValidation()
        {
            foreach (var pageUserControl in Children)
            {
                pageUserControl.PropertyChanged -= PageUserControlOnPropertyChanged;

                var page = pageUserControl as IValidationUpdate;
                if (page != null) page.UpdateValidation();

                pageUserControl.PropertyChanged += PageUserControlOnPropertyChanged; ;
            }

            UpdateErrors();
        }

        private void PageUserControlOnPropertyChanged(object sender, System.ComponentModel.PropertyChangedEventArgs propertyChangedEventArgs)
        {
            UpdateErrors();
        }

        private void UpdateErrors()
        {
            HasErrors = Children.Any(x => x.HasErrors);
            Errors = HasErrors ? "Есть ошибки" : null;
        }
    }
}
