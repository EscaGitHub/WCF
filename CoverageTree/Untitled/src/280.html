namespace Swr.Configurator.View.Settings.Pages
{
    /// <summary>
    /// Interaction logic for SpecificationTypeTab.xaml
    /// </summary>
    public partial class SpecificationTypeTab
    {
        public static readonly DependencyProperty SettingsProperty =
            DependencyProperty.Register("Settings", typeof(SpecificationTypeVM), typeof(SpecificationTypeTab), new PropertyMetadata(default(SpecificationTypeVM)));

        public static readonly DependencyProperty PageViewModelProperty =
            DependencyProperty.Register("PageViewModel", typeof(SpecificationTypesVM), typeof(SpecificationTypeTab), new PropertyMetadata(default(SpecificationTypesVM)));

        public SpecificationTypeTab()
        {
            InitializeComponent();

            
        }

        public SpecificationTypeVM Settings
        {
            get { return (SpecificationTypeVM)GetValue(SettingsProperty); }
            set { SetValue(SettingsProperty, value); }
        }

        public SpecificationTypesVM PageViewModel
        {
            get { return (SpecificationTypesVM)GetValue(PageViewModelProperty); }
            set { SetValue(PageViewModelProperty, value); }
        }

        private void Selector_OnSelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            if (sender is ComboBox comboBox)
            {
                if (comboBox.DataContext is ComponentTypeAttributeVM componentRow)
                {
                    if (string.IsNullOrWhiteSpace(componentRow.TableAlias))
                    {
                        componentRow.TableAlias = componentRow.Attribute.AttributeName;
                    }
                }
            }
        }
    }
}
