using System;
using System.Collections.Generic;
using System.ComponentModel.DataAnnotations;

namespace Swr.Configurator.Common.Data.Api
{
    public class ComponentApiExist : IComponentApi
    {
        public ComponentApiExist()
        {
            Attributes = new List<AttributeApi>();
        }

        [CannotBeEmptyGuid(ErrorMessage = "Field 'Id' is required.")]
        public Guid Id { get; set; }

        /// <summary>
        /// Дата создания
        /// </summary>
        public DateTime CreateDate { get; set; }

        /// <summary>
        /// Дата изменения
        /// </summary>
        public DateTime ChangeDate { get; set; }

        [Required(ErrorMessage = "Field 'TypeName' is required.")]
        public string TypeName { get; set; }

        [CannotBeEmpty(ErrorMessage = "Field 'Attributes' is required.")]
        public List<AttributeApi> Attributes { get; set; }
        public int Counter { get; set; }

        /// <summary>
        /// Имеет ли компонент дочерние элементы
        /// </summary>
        public bool HasChildren { get; set; }

       
    }
}
