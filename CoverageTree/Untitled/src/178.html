using System;
using System.Collections.Generic;
using System.Text;
using Swr.Configurator.Common.Settings.Columns;

namespace Swr.Configurator.Common.Data.ObjectModel
{
    public class AttributeForTypeObject
    {
        /// <summary>
        /// 
        /// </summary>
        public Guid Id { get; set; }

        /// <summary>
        /// Ссылка на атрибут из общего списка атрибутов
        /// </summary>
        public AttributeObject Attribute { get; set; }

        /// <summary>
        /// Имя атрибута для отображения на карте
        /// </summary>
        public string CardAlias { get; set; }

        /// <summary>
        /// Обязательность
        /// </summary>
        public bool IsNecessary { get; set; }

        /// <summary>
        /// Позиция на карте данных
        /// </summary>
        public int CardPosition { get; set; }

        /// <summary>
        /// Значение атрибута
        /// </summary>
        public string Value { get; set; }

        /// <summary>
        /// Маска имени
        /// </summary>
        public bool IsMask { get; set; }

        /// <summary>
        /// Тип колонки в спецификации
        /// </summary>
        public ColumnType ColumnType { get; set; }

        /// <summary>
        /// Позиция в таблице спецификации
        /// </summary>
        public int TablePosition { get; set; }

        /// <summary>
        /// Имя атрибута для отображения в таблице спецификации
        /// </summary>
        public string TableAlias { get; set; }

        /// <summary>
        /// Признак удаленного атрибута типа
        /// </summary>
        public bool IsDeleted { get; set; }

        public void Dispose()
        {
            Attribute = null;
        }
    }
}
