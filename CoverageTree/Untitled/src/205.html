using System.Collections.Generic;
using System.Linq;
using Swr.Configurator.Common.Data;

namespace Swr.Configurator.Data
{
	internal class PageElement
	{
		public IEnumerable<INameElement> Elements { get; }

		public PageElement(IEnumerable<INameElement> elements, int startNumber)
		{
			Elements = elements;
			StartNumber = startNumber;
		}

		public int StartNumber { get; }

		public int EndNumber
		{
			get => StartNumber + Elements.Count() - 1;
		}
	}
}
