using System;
using System.Collections.Generic;
using System.Net;
using System.Security.Claims;
using System.Text;
using Microsoft.AspNetCore.Http;

namespace Swr.Configurator.Test
{
    public static class HttpContextMock
    {
        public static DefaultHttpContext Get()
        {
            var user = new ClaimsPrincipal(new ClaimsIdentity(new Claim[]
            {
                new Claim(ClaimTypes.Name, "UserForTests"),
                new Claim("username", "UserForTests"),

            }, "mock"));

            var defContext = new DefaultHttpContext { User = user };
            defContext.Connection.RemoteIpAddress = IPAddress.Parse("127.0.0.1");

            defContext.Request.Host = new HostString("localhost");
            defContext.Request.PathBase = PathString.Empty;
            defContext.Request.Path = PathString.Empty;
            defContext.Request.QueryString = QueryString.Empty;

            return defContext;
        }
    }
}
