using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using Microsoft.VisualStudio.TestTools.UnitTesting;
using Swr.Configurator.Data;
using Swr.Configurator.WebApi;

namespace Swr.Configurator.Test.Net
{
    [TestClass]
    public class IdentityClientTests
    {
        [ClassInitialize]
        public static void InitializeClass(TestContext context)
        {
            StaticConnectionSettings.IdentityAddress = "https://skobelev:444";
            StaticConnectionSettings.BaseApiAddress = "https://localhost:44339";
        }

        [TestMethod]
        public void GetRoleTest()
        {
            var  task = IdentityClient.GetToken("admin", "123");

            var token = task.Result;

            Assert.IsNotNull(token);

            var task1 = IdentityClient.GetUserInfo();

            var userInfo = task1.Result;

            Assert.IsNotNull(userInfo);

            var role = userInfo.FirstOrDefault(t => t.Type == "role");

            Assert.IsNotNull(role);
            Assert.AreEqual("администратор", role.Value);
        }
    }
}
