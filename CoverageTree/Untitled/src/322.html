using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Windows.Documents;
using SwrUILogic.Data;

namespace Swr.Configurator.View.Settings.ViewModels
{
    public class AuthorizeTypeSettingsVm : ErrorCollector, IDataErrorInfo
    {
        private ComponentTypeVM _componentType;
        private SpecificationTypeVM _specificationType;

        public ComponentTypeVM ComponentType
        {
            get => _componentType;
            set
            {
                _componentType = value;

                OnPropertyChanged("ComponentType");
            }
        }

        public SpecificationTypeVM SpecificationType
        {
            get => _specificationType;
            set
            {
                _specificationType = value;

                OnPropertyChanged("SpecificationType");
            }
        }


        public override void UpdateValidation()
        {
            CalculateErrors(null);
        }

        public string this[string propertyName]
        {
            get
            {
                CalculateErrors(propertyName);

                if (string.IsNullOrWhiteSpace(propertyName))
                {
                    return string.Join(Environment.NewLine, Errors.Select(x => x.ErrorMessage));
                }
                else
                {
                    return string.Join(Environment.NewLine, Errors.Where(x => x.PropertyName == propertyName).Select(x => x.ErrorMessage));
                }
            }
        }

        private void CalculateErrors(string propertyName)
        {
            if (String.IsNullOrWhiteSpace(propertyName) || propertyName == "ComponentType")
            {
                var name = "ComponentType";

                DeleteErrorsForProperty(this, name);

                if (ComponentType == null) AddError(this, name, "Тип компонента не должен быть пустым");
            }

            if (String.IsNullOrWhiteSpace(propertyName) || propertyName == "SpecificationType")
            {
                var name = "SpecificationType";

                DeleteErrorsForProperty(this, name);

                if (SpecificationType == null) AddError(this, name, "Тип спецификации не должен быть пустым");
            }
        }

        public string Error => this[null];
    }
}
