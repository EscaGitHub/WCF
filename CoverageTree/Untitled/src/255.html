using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Swr.Configurator.Logic.Checker
{
	public class EmptyChecker: IChecker
	{
		public (bool hasError, string errorText) Check(string value)
		{
            if(string.IsNullOrWhiteSpace(value)) return (true, "Значение не может быть пустым");

            return (false, string.Empty);
        }
	}
}
