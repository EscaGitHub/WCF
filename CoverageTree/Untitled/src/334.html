using System.Windows;
using System.Windows.Media;

namespace Swr.Configurator.View.Controls
{
	public partial class VectorIconUC
	{
		public static readonly DependencyProperty IconColorProperty = DependencyProperty.Register("IconColor", typeof (SolidColorBrush), typeof (VectorIconUC), new PropertyMetadata(default(SolidColorBrush)));
		public static readonly DependencyProperty DataProperty = DependencyProperty.Register("Data", typeof (string), typeof (VectorIconUC), new PropertyMetadata(string.Empty));
		public static readonly DependencyProperty DisabledIconColorProperty = DependencyProperty.Register("DisabledIconColor", typeof (SolidColorBrush), typeof (VectorIconUC), new PropertyMetadata(default(SolidColorBrush)));

		public static readonly DependencyProperty TopProperty = DependencyProperty.Register("Top", typeof (int), typeof (VectorIconUC), new PropertyMetadata(default(int)));
		public static readonly DependencyProperty LeftProperty = DependencyProperty.Register("Left", typeof (int), typeof (VectorIconUC), new PropertyMetadata(default(int)));

		public VectorIconUC()
		{
			InitializeComponent();
		}

		public SolidColorBrush IconColor
		{
			get { return (SolidColorBrush) GetValue(IconColorProperty); }
			set { SetValue(IconColorProperty, value); }
		}

		public string Data
		{
			get { return (string) GetValue(DataProperty); }
			set { SetValue(DataProperty, value); }
		}

		public SolidColorBrush DisabledIconColor
		{
			get { return (SolidColorBrush) GetValue(DisabledIconColorProperty); }
			set { SetValue(DisabledIconColorProperty, value); }
		}

		public int Top
		{
			get { return (int) GetValue(TopProperty); }
			set { SetValue(TopProperty, value); }
		}

		public int Left
		{
			get { return (int) GetValue(LeftProperty); }
			set { SetValue(LeftProperty, value); }
		}
	}
}
